Node.js / JavaScript:

                

                    
const response = await fetch('https://productdescriptions.io/api/v1/generate', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-API-Key': 'your_api_key_here'
  },
  body: JSON.stringify({
    products: [{
      title: 'Product Name',
      product_type: 'Category',
      vendor: 'Brand'
    }]
  })
});

const data = await response.json();
console.log(data.results);

                

                
                
Python:

                

                    
import requests

response = requests.post(
    'https://productdescriptions.io/api/v1/generate',
    headers={
        'X-API-Key': 'your_api_key_here',
        'Content-Type': 'application/json'
    },
    json={
        'products': [{
            'title': 'Product Name',
            'product_type': 'Category',
            'vendor': 'Brand'
        }]
    }
)

data = response.json()
print(data['results'])
